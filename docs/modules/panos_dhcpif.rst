Examples
--------

 ::

    
    # enable DHCP client on ethernet1/1 in zone public
    - name: configure ethernet1/1
      panos_dhcpif:
        password: "admin"
        ip_address: "192.168.1.1"
        if_name: "ethernet1/1"
        zone_name: "public"
        create_default_route: "yes"
